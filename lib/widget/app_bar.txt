import 'package:flutter/cupertino.dart';
import 'package:flutter/material.dart';
import 'package:color_game/controller/game_control.dart';

Widget AppBar({Color backgroundColor, Text title, List<Widget> actions, int elevation}){
  Game game=Game();
  return PreferredSize(
        child: AppBar(
          actions: <Widget>[
            IconButton(
              icon: Icon(
                Icons.settings,
                color: Colors.black,
                size: 30,
              ),
              onPressed: () {},
            ),
            IconButton(
              icon: Icon(
                Icons.refresh,
                color: Colors.black,
                size: 30,
              ),
              onPressed: () {
                game.setGameOver();
                _restart();
              },
            )
          ],
          title: Text(
            "Transparent AppBar",
            style: TextStyle(color: Colors.black),
          ),
          backgroundColor: Colors.transparent,
          elevation: 0,
        ),
        preferredSize: Size.fromHeight(70),
      );
}
